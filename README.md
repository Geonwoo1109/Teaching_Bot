--Json 다루는 방식--
무작정 넣으면 넣어지고, 무작정 지우면 지워진다!

var json = {"a":"b"};
json["c"] = "d"    // json = {"a":"b", "c":"d"}
delete json["a"]   // json = {"c":"d"}


--json 선언 차이--
.a는 a가 문자열일때
[a]는 a가 변수일때
